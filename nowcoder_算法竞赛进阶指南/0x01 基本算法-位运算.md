A	a^b：https://ac.nowcoder.com/acm/contest/996/A
-------------

```cpp
#include<iostream>
#include<algorithm>
using namespace std;

int main()
{
    std::ios::sync_with_stdio(false);
    cin.tie(0);cout.tie(0);
    long long int a,b,p;
    cin>>a>>b>>p;
    long long int ans=1;
    while(b)
    {
        if(b&1)    ans=ans*a%p;
        b>>=1;    //位运算每次b右移一位
        a=a*a%p;  //每次a要翻倍，且%p
    }
    cout<<ans%p;
}
```

C   64位整数乘法：https://ac.nowcoder.com/acm/contest/996/C
```cpp
#include<iostream>
#include<algorithm>
using namespace std;

int main()
{
    std::ios::sync_with_stdio(false);
    cin.tie(0);cout.tie(0);
    long long int a,b,p;
    cin>>a>>b>>p;
    long long int ans=0;
    while(b)
    {
        if(b&1)
        {
            ans=(ans+a)%p;
        }
        b>>=1;
        a=(a*2)%p;
    }
    cout<<ans<<endl;
}
//a*b=a*101010110...(b的二进制形式)
```
