A	a^b：https://ac.nowcoder.com/acm/contest/996/A
-------------

```cpp
#include<iostream>
#include<algorithm>
using namespace std;

int main()
{
    std::ios::sync_with_stdio(false);
    cin.tie(0);cout.tie(0);
    long long int a,b,p;
    cin>>a>>b>>p;
    long long int ans=1;
    while(b)
    {
        if(b&1)    ans=ans*a%p;
        b>>=1;    //位运算每次b右移一位
        a=a*a%p;  //每次a要翻倍，且%p
    }
    cout<<ans%p;
}
```

C   64位整数乘法：https://ac.nowcoder.com/acm/contest/996/C
------------

```cpp
#include<iostream>
#include<algorithm>
using namespace std;

int main()
{
    std::ios::sync_with_stdio(false);
    cin.tie(0);cout.tie(0);
    long long int a,b,p;
    cin>>a>>b>>p;
    long long int ans=0;
    while(b)
    {
        if(b&1)
        {
            ans=(ans+a)%p;
        }
        b>>=1;
        a=(a*2)%p;
    }
    cout<<ans<<endl;
}
//a*b=a*101010110...(b的二进制形式)
```
B	Raising Modulo Numbers：https://ac.nowcoder.com/acm/contest/996/B
------------------
```cpp
#include<iostream>
#include<algorithm>
using namespace std;

long long fastm(long long ai,long long bi,int MOD)
{
    long long res=1;
    while(bi)
    {
        if(bi&1)    res=(res*ai)%MOD;
        bi>>=1;
        ai=ai*ai%MOD;
    }
    return res;
}

int main()
{
    std::ios::sync_with_stdio(false);
    cin.tie(0);cout.tie(0);
    int t;cin>>t;
    while(t--)
    {
        int MOD;cin>>MOD;
        int n;cin>>n;
        long long ans=0;
        while(n--)
        {
            long long ai,bi;cin>>ai>>bi;
            ans=(ans+fastm(ai,bi,MOD))%MOD;
        }
        cout<<ans<<endl;
    }
}
```
